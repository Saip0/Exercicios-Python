while True:
    N1 = float(input("Digite Um Número:"))
    if N1 <= 10 and N1 > 0:
        print(f"{"="*5}{"Tabuada":^}{"="*5}")
        for I in range(1,11):
            print(f"{N1} X {I} = {N1 * I}")
        print(f"{"="*15}\n")
    elif N1 == 191918:
        break
    else:
        print("O Número Deve Ser Maior Que 0 e Menor Que 10")
